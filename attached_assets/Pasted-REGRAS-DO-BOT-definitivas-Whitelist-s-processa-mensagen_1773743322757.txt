  if (type !== 'notify') return

        msg.message?.extendedTextMessage?.text ??
        ''
      ).trim()

      if (!texto) continue

      const phone = jid.replace('@s.whatsapp.net', '')
      console.log(`[WhatsApp] ${phone}: "${texto}"`)

      // ── WHITELIST ─────────────────────────────────────
      const contato = await getWhitelistByPhone(phone)
      if (!contato || !contato.active) {
        console.log(`[WhatsApp] ${phone} não autorizado — silêncio`)
        continue
      }

      // ── INTERPRETAR ───────────────────────────────────
      const resultado = interpretarMensagem(texto)

      if (!resultado.shouldReply) {
        console.log(`[WhatsApp] "${texto}" — conversa normal, silêncio`)
        continue
      }

      // ── CONSULTAR ─────────────────────────────────────
      let resposta = ''
      let found = false

      if (resultado.tipo === 'urgente') {
        const r = await consultarUrgente()
        resposta = r.resposta
        found = r.found
      } else if (resultado.tipo === 'op') {
        const r = await consultarOp(resultado.codigo)
        resposta = r.resposta
        found = r.found
      } else if (resultado.tipo === 'material') {
        const r = await consultarMaterial(resultado.codigo)
        resposta = r.resposta
        found = r.found
      }

      if (!resposta) continue

      // ── RESPONDER ─────────────────────────────────────
      await sock?.sendMessage(jid, { text: resposta })
      console.log(`[WhatsApp] Resposta enviada para ${phone}`)

      // ── HISTÓRICO ─────────────────────────────────────
      await saveQueryHistory({
        phone,
        type: resultado.tipo,
        query: resultado.codigo,
        found,
        response: resposta,
      })

    } catch (err) {
      console.error('[WhatsApp] Erro ao processar mensagem:', err)
      // nunca derruba o servidor
    }
  }

typescriptinterface InterpretacaoResult {
  shouldReply: boolean
  tipo: 'material' | 'op' | 'urgente' | null
  codigo: string
}

export function interpretarMensagem(texto: string): InterpretacaoResult {
  const t = texto.trim()
  const NAO_RESPONDE = { shouldReply: false, tipo: null, codigo: '' } as const

  // ── Comando !urgente ──────────────────────────────────
  if (/^!urgente/i.test(t)) {
    return { shouldReply: true, tipo: 'urgente', codigo: '!urgente' }
  }

  // ── OP explícita: "OP 119443" ou "op119443" ──────────
  const opExplicita = t.match(/^op[\s\-]?(\d{4,})/i)
  if (opExplicita) {
    return { shouldReply: true, tipo: 'op', codigo: opExplicita[1] }
  }

  // ── Número puro com 6+ dígitos = OP ──────────────────
  // Ex: "119443", "  119443  "
  // Mensagem deve ser SÓ o número (sem texto junto)
  const somenteNumero = t.replace(/\s/g, '')
  if (/^\d{6,}$/.test(somenteNumero)) {
    return { shouldReply: true, tipo: 'op', codigo: somenteNumero }
  }

  // ── Código de material com ponto ou traço ─────────────
  // Ex: "50.01.0036", "10-AB-0012", "50.01.0036 status?"
  const comPontoTraco = t.match(/\b(\d{2,}[\.\-][A-Z0-9]{2,}[\.\-]?[A-Z0-9]*)\b/i)
  if (comPontoTraco) {
    return { shouldReply: true, tipo: 'material', codigo: comPontoTraco[1] }
  }

  // ── Código alfanumérico isolado (sem espaços, 4+ chars) ─
  // Ex: "MAT0036", "AB1234" — mas NÃO palavras comuns
  // A mensagem inteira deve ser o código (sem texto explicativo)
  if (/^[A-Z]{1,4}\d{3,}$/i.test(t) || /^\d{3,}[A-Z]{1,4}$/i.test(t)) {
    return { shouldReply: true, tipo: 'material', codigo: t }
  }

  // ── Todo o resto = conversa normal ───────────────────
  // "oi", "tudo bem?", "qual o prazo?", "preciso de info"
  // → silêncio absoluto
  return NAO_RESPONDE
}
